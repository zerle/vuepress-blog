    webpack5--从零配置一个基础项目

    2022/4/21 Webpack5

    # 第一部分： 初始化项目

    源码请戳这里 (opens new window) 目标：完成一个能够打包运行的最基础配置，为后续逐步完善做准备。 示例的基础目录结构如下

    # 创建项目目录、初始化 package.json、安装Webpack 并添加配置文件 webpack.config.js

    本文使用yarn 安装项目依赖

    # 创建文件夹，并进入
mkdir webpack5-demo && cd webpack5-demo

# 全局安装yarn
npm install --global yarn

# 初始化package.json
yarn init --yes

# 安装webpack webpack-cli 到开发依赖
# webpack命令执行之后，会寻找webpack-cli这个包并执行，详见入口文件的逻辑：node_modules/webpack/bin/webpack.js
yarn add webpack webpack-cli -D

    根目录创建 webpack.config.js 文件并配置 entry 和 output

    const path = require('path')

module.exports = {
  entry: path.resolve(__dirname, './src/main.js'),
  output: {
    filename: './js/[name].[hash:8].js',
    path: path.resolve(__dirname, './dist')
  },
  mode: 'production'
}

    # CSS相关依赖-- css-loader、style-loader

    # css-loader 处理.css文件; style-loader 将样式通过<style>标签的形式挂载到页面的head部分
yarn add css-loader style-loader -D

    # 配置webpack.config.js
module: {
  rules: [
    {
      test: /\.css$/,
      use: ['style-loader', 'css-loader']
    }
  ]
}

    # HTML模板-- html-webpack-plugin

    #  在打包结束后，自动生成一个html文件，并把打包生成的js文件引入到这个html文件当中
yarn add html-webpack-plugin -D

    根目录新建public文件夹, public目录下新建index.html

    # webpack.config.js
plugins: [
  new HtmlWebpackPlugin({
    template: path.join(__dirname, './public/index.html'),
    filename: 'index.html'
  })
]

    # package.json中配置打包命令
"scripts": {
  "build": "webpack"
}
# 执行 yarn build后,dist目录为打包后的文件
yarn build

    # 清空打包文件

    # webpack.config.js  output中配置clean:true
output: {
    // ...
    clean: true
  },

    webpack5.20之前 需要安装插件 clean-webpack-plugin

    # 本地开启服务-- webpack-dev-server

    yarn add  webpack-dev-server -D

    # webpack.config.js
module.exports = {
  // ...
  devServer: {
    open: true,
    port: 8888,
    hot: true
  }
}
# package.json 增加本地启动服务命令
"scripts": {
    // ...
    "dev": "webpack serve"
  },

    # 拆分webpack.config.js区分环境

    目标：为了方便开发调试和上线，需要将webpack.config.js拆分为以下三个文件。

    webpack.dev.config.js：开发环境配置 webpack.prod.config.js：生产环境配置 webpack.base.config.js：开发和生产环境配置的共享配置

    开发环境和生产环境的区别

    生产环境可能需要分离CSS成单独文件，以便多个页面共享同一个CSS文件 生产环境需要压缩 HTML/CSS/JS 代码 开发环境需要 SourceMap 文件方便调试 开发环境需要HMR、devServer等功能 ......

    总的来说：开发环境帮助我们快速开发测试联调，生产环境保证上线环境的打包流程是最优化体验。

    # 安装webpack-merge

    # 用于连接数组，合并对象
yarn add webpack-merge -D 

    # demo
merge( 
    {a : [1],b:5,c:20}, 
    {a : [2],b:10, d: 421} 
)
//合并后结果
{a : [1,2] ,b :10 , c : 20, d : 421}

    根目录新建build文件夹，build目录下新建 webpack.dev.config.js webpack.prod.config.js webpack.base.config.js

    # webpack.base.config.js
const path = require('path')
const HtmlWebpackPlugin = require('html-webpack-plugin')

module.exports = {
  entry: path.resolve(__dirname, '../src/main.js'),
  output: {
    filename: './js/[name].[hash:8].js',
    path: path.resolve(__dirname, '../dist'),
    clean: true
  },
  module: {
    rules: [
      {
        test: /\.css$/,
        use: ['style-loader', 'css-loader']
      }
    ]
  },
  plugins: [
    new HtmlWebpackPlugin({
      template: path.join(__dirname, '../public/index.html'),
      filename: 'index.html'
    })
  ]
}

# webpack.dev.config.js

const { merge } = require('webpack-merge')
const webpackBaseConfig = require('./webpack.base.config')

module.exports = merge(webpackBaseConfig, {
  mode: 'development',
  devServer: {
    open: true,
    port: 8888,
    hot: true
  }
})

# webpack.prod.config.js

const { merge } =  require('webpack-merge')
const webpackBaseConfig = require('./webpack.base.config')

module.exports = merge(webpackBaseConfig, {
  mode: 'production'
})

    # 环境变量

    使用 cross-env，跨操作系统设置 node 环境变量。

    yarn add cross-env -D

    "scripts": {
  "build": "cross-env NODE_ENV=production webpack --config ./build/webpack.prod.config.js --progress --color",
  "dev": "cross-env NODE_ENV=development webpack serve --config ./build/webpack.dev.config.js --progress --color"
},

    # sourcemap

    sourcemap 产生的文件映射了压缩后的代码所对应的转换前的源代码位置，解决了代码压缩后难以调试的问题。 具体可参照文档，根据需要配置参数值，不同的值会明显影响到构建(build)和重新构建(rebuild)的速度。

    // webpack.dev.js
devtool: 'eval-cheap-module-source-map', 
// webpack.prod.js
devtool: 'cheap-module-source-map',

    # ES6转ES5 -- babel-loader

    yarn add babel-loader @babel/core @babel/preset-env @babel/plugin-transform-runtime -D
yarn add @babel/runtime @babel/runtime-corejs3 -D

    根目录新建.babelrc文件，配置如下

    {
  "presets": ["@babel/preset-env"],
  "plugins": [
      [
          "@babel/plugin-transform-runtime",
          {
              "corejs": 3
          }
      ]
  ]
}
# webpack.base.config.js 中配置如下
// ...
module.exports = {
  // ...
  module: {
    rules: [
      // ...
      {
        test: /\.js$/,
        use: 'babel-loader',
        // excluder: /node_modules/,
        include: path.resolve('src')
      }
    ]
  },
  // ...
}

    # 配置alias别名

    创建 import 或 require 的别名，来确保模块引入变得更简单

    // webpack.base.config.js
resolve: {
	alias: {
		"@": path.resolve(__dirname, "../src"),
		assets: path.resolve(__dirname, '../src/assets/'),
	},
}

    # 处理图片等静态资源

    Webpack5 之前我们处理静态资源比如PNG 图片、SVG 图标等等，需要用到url-loader，file-loader，raw-loader。Webpack5 提供了内置的静态资源构建能力，我们不需要安装额外的 loader，仅需要简单的配置就能实现静态资源的打包和分目录存放。这三个loader在github上也停止了更新。 webpack5使用四种新增的资源模块（Asset Modules）替代了这些loader的功能。 asset/resource 将资源分割为单独的文件，并导出url，就是之前的 file-loader的功能. asset/inline 将资源导出为dataURL（url(data:)）的形式，之前的 url-loader的功能. asset/source 将资源导出为源码（source code）. 之前的 raw-loader 功能. asset 自动选择导出为单独文件或者 dataURL形式（默认为8KB）. 之前有url-loader设置asset size limit 限制实现。

    //webpack.base.config.js
{
  test: /\.(png|svg|jpg|gif)$/,
  include: [path.resolve('src')],
  type: 'asset',
  parser: {
    dataUrlCondition: {
      maxSize: 8 * 1024
    }
  },
  generator: {
    filename: "img/[name].[hash:6][ext]"
  }
}

    # 静态资源输出到根目录 -- copy-webpack-plugin

    yarn add copy-webpack-plugin -D

    当某些文件不需要经过webpack打包处理而直接使用，这里我们可以使用 copy-webpack-plugin 这个插件，在构建的时候，将 public/ 的静态资源直接复制到 dist根目录下

    # webpack.prod.config.js
new CopyWebpackPlugin({
  patterns: [
    {
      from: path.resolve(__dirname, '../public/js'),
      to: path.resolve(__dirname, '../dist/public/js'),
    }
  ]
}),

    # 提取样式文件 -- mini-css-extract-plugin

    本插件会将 CSS 提取到单独的文件中，为每个包含 CSS 的 JS 文件创建一个 CSS 文件，并且支持 CSS 和 SourceMaps 的按需加载。

    yarn add mini-css-extract-plugin -D

    # webpack.prod.config.js
plugins: [
//...
	new miniCssExtractPlugin({
		filename: './css/[name].[contenthash].css',
		chunkFilename: './css/[id].[contenthash].css',
	})
]

# webpack.prod.config.js
rules: [
  {
    test: /\.css/,
    use: [
      MiniCssExtractPlugin.loader,
      'css-loader',
      'postcss-loader'
    ]
  }
]
# webpack.dev.config.js 
# 删除webpack.base.config.js 中css的配置
module: {
  rules: [
    {
      test: /\.css$/,
      use: ['style-loader', 'css-loader']
    }
  ]
},

    # 样式自动添加浏览器前缀 -- PostCSS 和 autoprefixer

    yarn add postcss-loader postcss autoprefixer -D 

    # webpack.dev.config.js 和 # webpack.prod.config.js 修改如下
{
  test: /\.css/,
  use: [
    // ...
    'postcss-loader'
  ]
}

# 根目录新建postcss.config.js，配置如下
module.exports = {
  plugins: [
    require('autoprefixer')({
      overrideBrowserslist: [
        "last 15 versions"
      ]
    })
  ]
}

    # CSS 压缩 -- css-minimizer-webpack-plugin

    yarn add css-minimizer-webpack-plugin -D
# webpack.prod.config.js
const CssMinimizerPlugin = require('css-minimizer-webpack-plugin');

module.exports = {
  # // 告知 webpack 使用 TerserPlugin 或其它在 optimization.minimizer 定义的插件压缩 bundle。
  optimization: {   
    minimizer: [
      // `...`,
      new CssMinimizerPlugin(),
    ],
  },
};
